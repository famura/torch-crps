5def crps_analytical_naive_integral( 

6 q: Distribution, 

7 y: torch.Tensor, 

8 x_min: float = -1e2, 

9 x_max: float = 1e2, 

10 x_steps: int = 5001, 

11) -> torch.Tensor: 

12 """Compute the Continuous Ranked Probability Score (CRPS) using the naive integral method. 

13 

14 Note: 

15 This function is not differentiable with respect to `y` due to the indicator function. 

16 

17 Args: 

18 q: A PyTorch distribution object, typically a model's output distribution. 

19 y: Observed values, of shape (num_samples,). 

20 x_min: Lower limit for integration for the probability space. 

21 x_max: Upper limit for integration for the probability space. 

22 x_steps: Number of steps for numerical integration. 

23 

24 Returns: 

25 CRPS values for each observation, of shape (num_samples,). 

26 """ 

27 

28 def integrand(x: torch.Tensor) -> torch.Tensor: 

29 """Compute the integrand $F(x) - 1(y <= x))^2$ to be used by the torch integration functions.""" 

30 cdf_value = q.cdf(x) 

31 indicator = (y_expanded <= x).float() 

32 return (cdf_value - indicator) ** 2 

33 

34 # Set integration limits. 

35 x_values = torch.linspace( 

36 start=torch.tensor(x_min, dtype=y.dtype, device=y.device), 

37 end=torch.tensor(x_max, dtype=y.dtype, device=y.device), 

38 steps=x_steps, 

39 dtype=y.dtype, 

40 device=y.device, 

41 ) 

42 

43 # Reshape for proper broadcasting. 

44 x_values = x_values.unsqueeze(-1) # shape: (x_steps, 1) 

45 y_expanded = y.unsqueeze(0) # shape: (1, num_samples) 

46 

47 # Compute the integral using the trapezoidal rule. 

48 integral_values = integrand(x_values) 

49 crps_values = torch.trapezoid(integral_values, x_values.squeeze(-1), dim=0) 

50 

51 return crps_values

54def crps_analytical_normal( 

55 q: Normal, 

56 y: torch.Tensor, 

57) -> torch.Tensor: 

58 """Compute the analytical CRPS assuming a normal distribution. 

59 

60 See Also: 

61 T. Gneiting et al., "Calibrated Probabilistic Forecasting Using Ensemble Model Output Statistics and Minimum 

62 CRPS Estimation", 2004, 

63 https://journals.ametsoc.org/view/journals/mwre/133/5/mwr2904.1.pdf 

64 Equation (5) for the analytical formula for CRPS of Normal distribution. 

65 

66 Args: 

67 q: A PyTorch distribution object, typically a model's output distribution. 

68 y: Observed values, of shape (num_samples,). 

69 

70 Returns: 

71 CRPS values for each observation, of shape (num_samples,). 

72 """ 

73 # Compute standard normal CDF and PDF. 

74 z = (y - q.loc) / q.scale # standardize 

75 standard_normal = torch.distributions.Normal(0, 1) 

76 phi_z = standard_normal.cdf(z) # Φ(z) 

77 pdf_z = torch.exp(standard_normal.log_prob(z)) # φ(z) 

78 

79 # Analytical CRPS formula. 

80 crps = q.scale * (z * (2 * phi_z - 1) + 2 * pdf_z - 1 / torch.sqrt(torch.tensor(torch.pi))) 

81 

82 return crps
